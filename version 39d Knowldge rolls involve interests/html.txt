  // Auto-calculate SOS as half of Max HP
  on("change:hp_max", function() {
    getAttrs(["hp_max"], function(values) {
      let hpMax = parseInt(values.hp_max) || 0;
      let sos = Math.floor(hpMax / 2);
      setAttrs({ sos: sos });
    });
  });
</script>

<!-- UPDATED SCRIPT: Core Ability TOTALS + NO STACKING ENFORCEMENT -->
<script type="text/worker">
const statsList = ["vig", "agi", "vit", "mnd", "spr", "per"];
const ambKeys = ["amb1", "amb2", "amb3", "amb4"];
const ambThresholds = {
  amb1: [1, 7, 12, 17],
  amb2: [5, 9, 14, 18],
  amb3: [7, 13, 19],
  amb4: [10, 15, 20]
};

on("change:stat_package change:package_bonus1_stat change:package_bonus2_stat change:package_bonus3_stat change:package_bonus4_stat change:package_bonus5_stat change:package_bonus6_stat change:amb1_stat change:amb2_stat change:amb3_stat change:amb4_stat change:racial_bonus_stat change:level", function() {
  getAttrs([
    "stat_package",
    "package_bonus1_stat","package_bonus2_stat","package_bonus3_stat",
    "package_bonus4_stat","package_bonus5_stat","package_bonus6_stat",
    "amb1_stat","amb2_stat","amb3_stat","amb4_stat",
    "racial_bonus_stat",
    "level"
  ], function(v) {
    const level = parseInt(v.level) || 1;
    const packageType = v.stat_package || "specialist";

    // Package bonus values
    let bonusValues = [];
    if (packageType === "generalist") bonusValues = [1,1,1,1,1,1];
    else if (packageType === "specialist") bonusValues = [3,2,2,-1,-1,-1];
    else if (packageType === "versatile") bonusValues = [2,2,2,0,0,-1];
    else bonusValues = [0,0,0,0,0,0];

    // Collect chosen stats for package slots
    let chosen = [];
    for (let i = 1; i <= 6; i++) {
      const stat = v[`package_bonus${i}_stat`] || "none";
      if (stat !== "none") chosen.push(stat);
    }

    // ENFORCE NO STACKING: if any duplicates, package bonuses are ignored (set to 0)
    const uniqueCount = new Set(chosen).size;
    const isValid = uniqueCount === chosen.length;

    let packageBonuses = { vig: 0, agi: 0, vit: 0, mnd: 0, spr: 0, per: 0 };
    if (isValid) {
      for (let i = 1; i <= 6; i++) {
        const chosenStat = v[`package_bonus${i}_stat`] || "none";
        if (chosenStat !== "none" && packageBonuses.hasOwnProperty(chosenStat)) {
          packageBonuses[chosenStat] += bonusValues[i - 1];
        }
      }
    }
    // If stacking detected, packageBonuses stay at 0 (no bonuses applied)

    // AMB level-based bonuses
    let ambBonuses = { vig: 0, agi: 0, vit: 0, mnd: 0, spr: 0, per: 0 };
    ambKeys.forEach(key => {
      const chosenStat = v[`${key}_stat`] || "none";
      if (chosenStat !== "none" && ambBonuses.hasOwnProperty(chosenStat)) {
        const thresholds = ambThresholds[key];
        const count = thresholds.filter(th => th <= level).length;
        ambBonuses[chosenStat] += count;
      }
    });

    // Racial +1
    const racialStat = v.racial_bonus_stat || "none";
    if (racialStat !== "none" && ambBonuses.hasOwnProperty(racialStat)) {
      ambBonuses[racialStat] += 1;
    }

    // Final totals
    let totals = {};
    statsList.forEach(stat => {
      totals[stat] = packageBonuses[stat] + ambBonuses[stat];
    });

    setAttrs(totals);
  });
});
</script>
